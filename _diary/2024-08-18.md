version: argoproj.io/v1alpha1
kind: Application
metadata:
  name: my-app
  namespace: argocd
spec:
  project: default # in-cluster
  syncPolicy:
    automated:
      prune: true
      selfHeal: false
      # ...
    syncOptions:
      - Key=Value
  source: # v2.5-
    repoUrl: "http://git-or-helm-repo"
    targetRevision: "branch-or-chart-version"
    # for git
    directory:
      path:
      include:
      exclude:
    # for helm
    chart: "helm-chart"
    helm:
      values: |
        key: value
        ...
  sources: # v2.6+
    - repoUrl: "http://helm-chart-repo"
      targetRevision: "helm-chart-version"
      chart: "helm-chart-name"
      helm:
        valueFiles:
          - "$values/path/of/values.yaml"
    
  desdination:
    name: "name-for-what"
    server: "kube-api-server-url"
    namesapce: "target-name-space"
